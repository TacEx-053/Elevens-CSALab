        import java.lang.Math

        public static String flip()
        {
                //because odds are 2:1 then a number will need to be genereated
                //between 1 and 3. If the number is 1 or 2 then heads otherwise 
                //tails.

                int num = (int) (Math.Random() * (3));

                if(num == 1 || num == 2)
                {
                        return "Heads";
                }
                if(num == 3)
                {
                        return "Tails";
                }
        }
